### 2\. Stack de Integración RDM‑MAP‑2D (Extracción)

```typescript
// frontend/rdm-map-2d.ts
import mapboxgl from "mapbox-gl";

mapboxgl.accessToken = process.env.MAPBOX_TOKEN ?? "";

const map = new mapboxgl.Map({
  container: "rdm-map-2d",
  style: "mapbox://styles/mapbox/dark-v11",
  center: [-98.667, 20.135], // RDM Nodo Zero
  zoom: 13.8,
  pitch: 45,
  bearing: -10,
});

map.on("load", () => {
  map.addSource("rdm-pois", {
    type: "geojson",
    data: "/vault/poi_nodes.json",
  });

  map.addLayer({
    id: "rdm-pois-layer",
    type: "circle",
    source: "rdm-pois",
    paint: {
      "circle-radius": 5,
      "circle-color": "#E2E8F0", // Platinum highlight
      "circle-stroke-width": 1.5,
      "circle-stroke-color": "#020617",
    },
  });
});
```
