### 🛰️ Generación de Grids de Soberanía (PyGMT)

```python
# pygmt/scripts/generate_tactic_grids.py
import pygmt
import xarray as xr
import rasterio
from rasterio.transform import from_bounds
from rdm_tos.core.config import REGION_ZERO, RESOLUTION_HIGH


def to_geotiff(grid: xr.DataArray, out_path: str) -> None:
    """
    Exporta un grid PyGMT/Xarray a GeoTIFF con metadatos geoespaciales completos.
    """
    lon, lat = grid.lon.values, grid.lat.values
    transform = from_bounds(
        float(lon.min()), float(lat.min()),
        float(lon.max()), float(lat.max()),
        grid.sizes["lon"], grid.sizes["lat"],
    )
    data = grid.values.astype("float32")
    height, width = data.shape

    with rasterio.open(
        out_path,
        "w",
        driver="GTiff",
        height=height,
        width=width,
        count=1,
        dtype="float32",
        crs="EPSG:4326",
        transform=transform,
    ) as dst:
        dst.write(data, 1)


def generate_sovereignty_grids() -> None:
    """
    Construye la base geoespacial táctica de Nodo Cero:
      - Relieve de alta resolución (DEM 15s)
      - Máscara tierra/mar (GSHHG-based)
      - Gradiente de pendiente (para rutas y riesgo)
    """
    print(f"🛰️ Generando grids para REGION_ZERO: {REGION_ZERO}")

    relief = pygmt.datasets.load_earth_relief(
        resolution=RESOLUTION_HIGH,
        region=REGION_ZERO,
        registration="gridline",
    )

    mask = pygmt.datasets.load_earth_mask(
        resolution=RESOLUTION_HIGH,
        region=REGION_ZERO,
    )

    slope = pygmt.grdgradient(
        grid=relief,
        radii="e15s",
        azimuth=0,
    )

    to_geotiff(relief, "pygmt/data/grids/rdm_relief_15s.tif")
    to_geotiff(slope,  "pygmt/data/grids/rdm_slope_15s.tif")

    xr.Dataset(
        {
            "relief": relief,
            "mask": mask,
            "slope": slope,
        }
    ).to_netcdf("pygmt/data/vault/rdm_tactic_base.nc")

    print("📂 Grids de soberanía exportados al Knowledge Vault.")


if __name__ == "__main__":
    generate_sovereignty_grids()
```

Estos artefactos no son cosmética.  
Son la **base dura** que alimenta:

- Capas analíticas de riesgo, accesibilidad y logística.  
- Simulación de rutas, despliegues y estrategias territoriales con contexto físico real.
